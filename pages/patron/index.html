Класс Patron

Patron на GitHub

Предназначен для превращения на лету гостя в патрона

Это поведение полезно в случаях, когда нужно из просто посетителя(гостя) сделать постоянного посетиеля(патрона). Это будет означать что если у источника данных есть новые данные, то источник отдаст эту новую порцию данных всем своим постоянным посетителям(патронам).

Пример 1.

import {
    Patron,
    GuestAware,
    Source,
    isPatronInPools,
    removePatronFromPools
} from 'patron-oop';

const source = new Source(1);
const timer = new GuestAware((guest) => {
    source.value(guest);
    const repeat = () => {
        if (!isPatronInPools(guest)) {
            return;
        }
        source.value((value) => {
            setTimeout(() => {
                source.give(value + 1);
                repeat();
            }, 1000);
        });
    }
    repeat();
})

const timerGuest = new Patron((value) => {
    const el = document.querySelector('.guest-result');
    if (el) {
        el.textContent = `Секунд прошло ${value}`;
    } else {
        removePatronFromPools(timerGuest);
    }
});
timer.value(timerGuest);
